
      {/* Hiển thị Date Picker */}
      {showDatePicker && (
        <DateTimePicker
          value={new Date()} // Ngày mặc định là ngày hiện tại
          mode="date" // Chế độ chọn ngày
          display={Platform.OS === "ios" ? "spinner" : "default"} // Giao diện tùy thuộc vào nền tảng

          maximumDate={new Date()} // Không cho phép chọn ngày trong tương lai
        />
      )}

      {/* Gợi ý */}
      <Text className="text-sm text-gray-500 mt-4">
        Bạn cần đủ 14 tuổi để sử dụng Zalo
      </Text>

      {/* Nút Chọn */}

        disabled={!birthday} // Vô hiệu hóa nếu chưa chọn ngày sinh
        className={`mt-10 py-3 rounded-lg ${
          birthday ? "bg-blue-500" : "bg-gray-300"
        }`}
      >
        <Text className="text-center text-white font-bold">Chọn</Text>
